Pixelator example

Draw example

pixelator.init('workspace-draw', {
    'size'      : 20,
    'fill'      : 'green',
    'border'    : 'none'
});
pixelator.draw(frog_matrix);

Switch example

pixelator.init('workspace-switch', {
    'size'      : 10,
    'fill'      : '#000',
    'border'    : '#fff'
});
pixelator.transform(frog_matrix, number_matrix);
